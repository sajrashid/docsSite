### Fetch Data

 ```js
    const [json, setJson] = useState([])
    useEffect(() => {
        async function fetchAPI() {
            const url = 'https://jsonplaceholder.typicode.com/docs'
            const response = await fetch(url)
            const json = await response.json()
            setJson(json)
        }
        fetchAPI()
      }, [])
 ```
